24/**
25 * @brief Realiza un diagnóstico sobre un puerto UART específico.
26 *
27 * El procedimiento:
28 * - Inicia el puerto UART a 9600 baudios.
29 * - Espera durante 3 segundos la llegada de datos.
30 * - Si recibe datos, los imprime por consola.
31 * - Si no recibe nada, informa que no hubo actividad.
32 *
33 * @param serialPort Referencia al objeto HardwareSerial (por ejemplo, Serial1 o Serial2).
34 * @param portName Nombre del puerto UART para mostrar en consola.
35 */
36void diagnoseUARTStream(HardwareSerial& serialPort, const char* portName) {
37 Serial.println("📡 Diagnosing " + String(portName));
38
39 serialPort.begin(9600);
40 delay(100);
41
42 Serial.println("⌛ Waiting for data on " + String(portName) + " for 3 seconds...");
43
44 unsigned long start = millis();
45 bool received = false;
46
47 while (millis() - start < 3000) {
48 if (serialPort.available()) {
49 received = true;
50 Serial.print("📥 Data received on ");
51 Serial.print(portName);
52 Serial.print(": ");
53 while (serialPort.available()) {
54 char c = serialPort.read();
55 Serial.print(c);
56 }
57 Serial.println();
58 break;
59 }
60 }
61
62 if (!received) {
63 Serial.println("🔻 No data detected on " + String(portName));
64 }
65
66 // Do not call serialPort.end() to keep the port active
67 Serial.println("✅ End of diagnostic.\n");
68}
