44/**
45 * @brief Calcula la salida del controlador PID.
46 *
47 * Implementa:
48 * - Proporcional
49 * - Integral con anti-windup condicional
50 * - Derivative-on-measurement (evita derivative kick)
51 * - Slew-rate limit (limitación de cambio por ciclo)
52 * - Saturación final
53 *
54 * @param setpoint Valor deseado.
55 * @param feedback Valor medido.
56 * @return Salida del controlador PID.
57 */
58float PID::calculoPID(float setpoint, float feedback, bool debugMSG)
59{
60 // 1) Error actual
61 this->error = setpoint - feedback;
62
63 // 2) Proporcional
64 this->proportional = this->Kp * this->error;
65
66 // Protección dt
67 if (this->dt <= 0.0f)
68 return this->output;
69
70 // 3) Derivada (derivative-on-measurement)
71 this->derivative = -(feedback - this->prevFeedback) / this->dt;
72
73 // 4) Anti-windup condicional
74 this->pidNoIntegral = this->proportional + this->Kd * this->derivative;
75 this->saturated = (this->pidNoIntegral > this->outputMax) || (this->pidNoIntegral < this->outputMin);
76
77 if (!this->saturated)
78 {
79 this->integral += this->error * this->dt;
80
81 if (this->integral > this->outputMax) this->integral = this->outputMax;
82 if (this->integral < this->outputMin) this->integral = this->outputMin;
83 }
84
85 // 5) PID completo sin limitación de cambio
86 this->rawOutput = this->proportional +
87 (this->Ki * this->integral) +
88 (this->Kd * this->derivative);
89
90 // 6) Slew-rate limit
91 this->deltaOutput = this->rawOutput - this->prevOutput;
92
93 if (this->deltaOutput > this->outputMaxChanged)
94 this->output = this->prevOutput + this->outputMaxChanged;
95 else if (this->deltaOutput < -this->outputMaxChanged)
96 this->output = this->prevOutput - this->outputMaxChanged;
97 else
98 this->output = this->rawOutput;
99
100 // 7) Saturación final
101 if (this->output > this->outputMax) this->output = this->outputMax;
102 if (this->output < this->outputMin) this->output = this->outputMin;
103
104 // 8) Guardar estado
105 this->prevError = this->error;
106 this->prevFeedback = feedback;
107 this->prevOutput = this->output;
108
109
110 // 9) Debug
111 if (debugMSG) printDebugPID(setpoint, feedback);
112
113 return this->output;
114
115};
