7/**
8 * @struct PinInfo
9 * @brief Representa un pin del sistema con nombre simbólico, número físico y familia funcional.
10 *
11 * Esta estructura permite describir cada pin de forma clara y tipada,
12 * facilitando diagnósticos, documentación y uso en tiempo de compilación.
13 *
14 * @param name Nombre simbólico del pin (ej. "PIN_PWM_10").
15 * @param number Número físico del pin en la placa Arduino.
16 * @param family Familia funcional (ej. "GPIO", "PWM", "ANALOG", "UART_RX").
17 */
18struct PinInfo {
19 const char* name; // Symbolic name of the pin
20 int number; // Physical pin number on the board
21 const char* family; // Pin family or category (e.g., "PWM", "GPIO")
22
23 /// Constructor constexpr para permitir uso en evaluaciones de tiempo de compilación
24 constexpr PinInfo(const char* n, int num, const char* fam) : name(n), number(num), family(fam) {}
